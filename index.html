print_example.py
# 输出文本
print("Hello, Python!")

# 输出变量

print(name)

# 输出多个值

# 使用格式化字符串
print(f"My name is {name} and I am {age} years old.")

# 输出到文件
with open("output.txt", "w") as f:

变量命名规则

                

                
                

                    
在Python中，变量命名需要遵循一定的规则，良好的命名习惯可以提高代码的可读性和可维护性。

                    
                    

                        
命名规则

                        
    
                            
  • 变量名只能包含字母、数字和下划线
    • 
                            
  • 变量名必须以字母或下划线开头
    • 
                            
  • 变量名区分大小写
    • 
                            
  • 不能使用Python关键字作为变量名
    •

# 有效的变量名

# 短路求值
print("\n短路求值示例:")
def true_func():
    print("调用了true_func")
    return True

def false_func():
    print("调用了false_func")
    return False

# and短路：第一个为False，不执行第二个
result = false_func() and true_func()
print(f"结果: {result}")

# or短路：第一个为True，不执行第二个
result = true_func() or false_func()
print(f"结果: {result}")

def calculate_area(radius):
    """
    计算圆的面积
    
    参数:
    radius (float): 圆的半径
    
    返回:
    float: 圆的面积
    """
    pi = 3.14159
    area = pi * radius ** 2
    return area

class Circle:
    """
    圆类，用于表示和操作圆形
    
    属性:
    radius (float): 圆的半径
    
    方法:
    calculate_area(): 计算圆的面积
    calculate_circumference(): 计算圆的周长
    """
    
    def __init__(self, radius):
        """
        初始化Circle对象
        
        参数:
        radius (float): 圆的半径
        """
        self.radius = radius
    
    def calculate_circumference(self):
        """计算圆的周长"""
        pi = 3.14159
        circumference = 2 * pi * self.radius
        return circumference

def check_eligibility(age, has_id, has_ticket):
    """检查是否有资格参加活动"""
    if age >= 18:
        if has_id:
            if has_ticket:
                return "可以参加活动"
            else:
                return "需要购买门票"
        else:
            return "需要出示有效身份证件"
    else:
        return "年龄不符合要求"

def calculate_discount(price, is_member, has_coupon):
    """计算折扣价格"""
    discount = 0
    
    if is_member:
        discount += 0.1  # 会员10%折扣
    
    if has_coupon:
        discount += 0.05  # 优惠券5%折扣
    
    # 最大折扣不超过20%
    discount = min(discount, 0.2)
    
    final_price = price * (1 - discount)
    return final_price, discount * 100

    count += 1

    attempts = 0

                print("太小了！再试一次。")

    attempts = 0
    max_attempts = 3

        username = input("请输入用户名: ")
        password = input("请输入密码: ")

            print("登录成功！")

    return False

# 计算圆的面积和周长
import math

def calculate_circle(radius):
    """
    计算圆的面积和周长
    
    参数:
    radius (float): 圆的半径
    
    返回:
    tuple: (面积, 周长)
    """
    area = math.pi * radius ** 2
    circumference = 2 * math.pi * radius
    return area, circumference

radius = 5

# 计算BMI
def calculate_bmi(weight, height):
    """
    计算BMI指数
    
    参数:
    weight (float): 体重（公斤）
    height (float): 身高（米）
    
    返回:
    tuple: (bmi值, 健康状况)
    """
    if height <= 0:
        return None, "身高必须大于0"
    
    bmi = weight / (height ** 2)
    
    if bmi < 18.5:
        status = "偏瘦"
    elif bmi < 24:
        status = "正常"
    elif bmi < 28:
        status = "超重"
    else:
        status = "肥胖"
    
    return bmi, status

weight = 70
height = 1.75
bmi, status = calculate_bmi(weight, height)
print(f"\n体重: {weight}kg, 身高: {height}m")
print(f"BMI: {bmi:.2f}, 健康状况: {status}")

# 计算列表统计信息
def calculate_statistics(numbers):
    """
    计算列表的统计信息
    
    参数:
    numbers (list): 数字列表

# 字符串处理函数
def process_string(text):
    """
    处理字符串，返回各种信息
    
    参数:
    text (str): 输入字符串
    
    返回:
    dict: 包含字符串信息的字典
    """
    if not isinstance(text, str):
        return None
    
    info = {
        "original": text,
        "length": len(text),
        "upper": text.upper(),
        "lower": text.lower(),
        "stripped": text.strip(),
        "words": text.split(),
        "word_count": len(text.split()),
        "vowel_count": sum(1 for char in text.lower() if char in 'aeiou')
    }
    
    return info

text = "  Hello, World! This is a test string.  "

    int: n的阶乘
    """

import datetime

# 计算年龄
def calculate_age(birth_date):
    """
    根据出生日期计算年龄
    
    参数:
    birth_date (datetime.date): 出生日期
    
    返回:
    int: 年龄
    """
    today = datetime.date.today()
    age = today.year - birth_date.year
    
    # 如果今年的生日还没过，年龄减1
    if today < datetime.date(today.year, birth_date.month, birth_date.day):
        age -= 1
    
    return age

def count_workdays(start_date, end_date):
    """
    计算两个日期之间的工作日数量（不包括周末）
    
    参数:
    start_date (datetime.date): 开始日期
    end_date (datetime.date): 结束日期
    
    返回:
    int: 工作日数量
    """
    if start_date > end_date:

    current_date = start_date
    
    while current_date <= end_date:
        # 0-4表示星期一到星期五
        if current_date.weekday() < 5:
            workdays += 1
        current_date += datetime.timedelta(days=1)
    
    return workdays

# 计算倒计时
def calculate_countdown(target_date):
    """
    计算到目标日期的倒计时
    
    参数:
    target_date (datetime.datetime): 目标日期时间
    
    返回:
    dict: 包含天、时、分、秒的倒计时字典
    """
    now = datetime.datetime.now()
    
    if target_date < now:
        return {"days": 0, "hours": 0, "minutes": 0, "seconds": 0}
    
    delta = target_date - now
    days = delta.days
    hours = delta.seconds // 3600
    minutes = (delta.seconds % 3600) // 60
    seconds = delta.seconds % 60
    
    return {
        "days": days,
        "hours": hours,
        "minutes": minutes,
        "seconds": seconds
    }

                writer.writerow(headers)
                writer.writerows(data)

        return True
    except Exception as e:

def read_csv_file(filename, as_dict=False):
    """
    读取CSV文件
    
    参数:
    filename (str): 文件名
    as_dict (bool): 是否以字典形式返回数据
    
    返回:
    tuple: (headers, data)，headers是表头列表，data是数据列表
    """
    try:
        with open(filename, "r", newline="", encoding="utf-8") as file:
            if as_dict:
                # 以字典形式读取
                reader = csv.DictReader(file)
                headers = reader.fieldnames
                data = list(reader)
            else:
                # 以列表形式读取
                reader = csv.reader(file)
                headers = next(reader)  # 获取表头
                data = list(reader)
        
        print(f"\nCSV文件 '{filename}' 读取成功！")
        return headers, data
    except FileNotFoundError:
        print(f"CSV文件 '{filename}' 不存在")
        return None, None
    except Exception as e:
        print(f"读取CSV文件时发生错误: {e}")
        return None, None

def process_csv_data(filename):

    处理CSV数据，进行简单的数据分析

    filename (str): 文件名

        return
    
    print(f"\n处理CSV数据 '{filename}':")
    
    # 计算平均年龄和平均GPA
    total_age = 0
    total_gpa = 0
    count = len(data)
    
    for row in data:
        total_age += int(row["age"])
        total_gpa += float(row["gpa"])
    
    avg_age = total_age / count
    avg_gpa = total_gpa / count
    
    print(f"学生总数: {count}")

    print(f"平均GPA: {avg_gpa:.2f}")

def count_word_frequency(text):
    """计算文本中每个单词出现的频率"""
    # 转换为小写并分割成单词
    words = text.lower().split()
    frequency = {}
    
    for word in words:
        # 去除标点符号
        cleaned_word = word.strip(".,!?;:()[]{}'\"")
        
        if cleaned_word:
            if cleaned_word in frequency:
                frequency[cleaned_word] += 1
            else:
                frequency[cleaned_word] = 1
    
    return frequency

def factorial(n):
    """计算n的阶乘"""
    if n < 0:
        return "阶乘只对非负整数定义"
    elif n == 0 or n == 1:
        return 1
    
    result = 1
    i = 2
    while i <= n:
        result *= i
        i += 1
    
    return result

def fibonacci_sequence(n):

        next_number = sequence[i-1] + sequence[i-2]
        sequence.append(next_number)
        i += 1
    
    return sequence

import math

def calculate_bmi(weight, height):
    """
    计算BMI指数
    
    参数:
    weight (float): 体重（公斤）
    height (float): 身高（米）
    
    返回:
    float: BMI指数
    """
    if height <= 0:
        raise ValueError("身高必须大于0")
    
    bmi = weight / (height ** 2)
    return round(bmi, 2)

def classify_bmi(bmi):
    """
    根据BMI指数分类
    
    参数:
    bmi (float): BMI指数
    
    返回:
    str: BMI分类
    """
    if bmi < 18.5:
        return "偏瘦"
    elif bmi < 24:
        return "正常"
    elif bmi < 28:
        return "超重"
    else:
        return "肥胖"

def calculate_compound_interest(principal, rate, years):
    """
    计算复利
    
    参数:
    principal (float): 本金
    rate (float): 年利率（百分比）
    years (int): 年数
    
    返回:
    float: 最终金额
    """
    rate_decimal = rate / 100
    amount = principal * (1 + rate_decimal) ** years
    return round(amount, 2)

def is_prime(n):
    """
    检查一个数是否为质数
    
    参数:
    n (int): 要检查的数
    
    返回:
    bool: 如果是质数返回True，否则返回False
    """
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    
    return True

def generate_primes(limit):
    """
    生成指定范围内的所有质数
    
    参数:
    limit (int): 上限
    
    返回:
    list: 质数列表
    """
    primes = []
    for num in range(2, limit + 1):
        if is_prime(num):
            primes.append(num)
    return primes

def count_workdays(start_date, end_date):
    """计算两个日期之间的工作日数"""
    workdays = 0
    current_date = start_date
    
    while current_date <= end_date:
        # 0是星期一，4是星期五
        if current_date.weekday() < 5:
            workdays += 1
        current_date += datetime.timedelta(days=1)
    
    return workdays

def calculate_age(birth_date):
    """根据出生日期计算年龄"""
    today = datetime.date.today()
    age = today.year - birth_date.year
    
    # 如果今年的生日还没过，年龄减1
    if today < datetime.date(today.year, birth_date.month, birth_date.day):
        age -= 1
    
    return age

def countdown(target_date):
    """计算到目标日期的倒计时"""
    today = datetime.date.today()
    delta = target_date - today
    
    if delta.days > 0:
        return f"距离 {target_date} 还有 {delta.days} 天"
    elif delta.days == 0:
        return f"今天就是 {target_date}!"
    else:
        return f"{target_date} 已经过去了 {-delta.days} 天"

def count_words_in_file(filename):
    """计算文件中的单词数量"""
    try:
        with open(filename, "r", encoding="utf-8") as file:

            return len(words)

        print(f"文件 {filename} 未找到！")
        return 0
    except Exception as e:
        print(f"读取文件时发生错误: {e}")
        return 0

def copy_file(source, destination):
    """复制文件"""
    try:
        with open(source, "rb") as src_file:
            with open(destination, "wb") as dest_file:
                # 分块复制，适合大文件
                while True:
                    chunk = src_file.read(1024 * 1024)  # 1MB块
                    if not chunk:
                        break
                    dest_file.write(chunk)
        print(f"文件 {source} 已复制到 {destination}")
        return True
    except FileNotFoundError:
        print(f"源文件 {source} 未找到！")
        return False
    except Exception as e:
        print(f"复制文件时发生错误: {e}")
        return False

def read_config_file(filename):
    """读取配置文件"""
    config = {}
    try:
        with open(filename, "r", encoding="utf-8") as file:
            for line in file:
                line = line.strip()
                # 跳过空行和注释
                if not line or line.startswith("#"):
                    continue
                
                # 解析键值对
                if "=" in line:
                    key, value = line.split("=", 1)
                    config[key.strip()] = value.strip()
        return config
    except FileNotFoundError:
        print(f"配置文件 {filename} 未找到！")
        return {}
    except Exception as e:
        print(f"读取配置文件时发生错误: {e}")
        return {}

def analyze_product_data(filename):
    """分析产品数据"""
    try:
        if not os.path.exists(filename):
            print(f"文件 {filename} 不存在！")
            return
        
        # 按类别统计产品数量和总价值
        category_stats = {}
        total_products = 0
        total_value = 0
        
        with open(filename, "r", encoding="utf-8") as file:
            reader = csv.DictReader(file)
            
            for row in reader:
                category = row["类别"]
                price = float(row["价格"])
                quantity = int(row["库存数量"])
                value = price * quantity
                
                # 更新类别统计
                if category not in category_stats:
                    category_stats[category] = {
                        "count": 0,
                        "total_value": 0
                    }
                
                category_stats[category]["count"] += 1
                category_stats[category]["total_value"] += value
                
                # 更新总体统计
                total_products += 1
                total_value += value
        
        # 显示统计结果
        print(f"\n产品数据分析结果:")
        print(f"总产品数量: {total_products}")
        print(f"总库存价值: ${total_value:,.2f}")
        print(f"\n按类别统计:")
        
        for category, stats in category_stats.items():
            print(f"{category}:")
            print(f"  产品数量: {stats['count']}")
            print(f"  总价值: ${stats['total_value']:,.2f}")
            print(f"  平均价值: ${stats['total_value'] / stats['count']:,.2f}")
        
        # 找出价值最高的产品
        print(f"\n价值最高的产品:")
        max_value_product = None
        max_value = 0
        
        with open(filename, "r", encoding="utf-8") as file:
            reader = csv.DictReader(file)
            
            for row in reader:
                price = float(row["价格"])
                quantity = int(row["库存数量"])
                value = price * quantity
                
                if value > max_value:
                    max_value = value
                    max_value_product = row
        
        if max_value_product:
            print(f"产品名称: {max_value_product['产品名称']}")
            print(f"产品ID: {max_value_product['产品ID']}")
            print(f"类别: {max_value_product['类别']}")
            print(f"价格: ${float(max_value_product['价格']):,.2f}")
            print(f"库存数量: {max_value_product['库存数量']}")
            print(f"总价值: ${max_value:,.2f}")
        
    except Exception as e:
        print(f"分析产品数据时发生错误: {e}")

def filter_products_by_category(filename, category, output_filename):
    """按类别筛选产品并保存到新文件"""
    try:
        if not os.path.exists(filename):
            print(f"文件 {filename} 不存在！")
            return False
        
        # 读取并筛选数据
        filtered_data = []
        headers = None
        
        with open(filename, "r", encoding="utf-8") as file:
            reader = csv.DictReader(file)
            headers = reader.fieldnames
            
            for row in reader:
                if row["类别"] == category:
                    filtered_data.append(row)
        
        if not filtered_data:
            print(f"没有找到类别为 '{category}' 的产品！")
            return False
        
        # 写入筛选后的数据
        with open(output_filename, "w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=headers)
            writer.writeheader()
            writer.writerows(filtered_data)
        
        print(f"\n已将类别为 '{category}' 的 {len(filtered_data)} 个产品保存到 {output_filename}")
        return True
    
    except Exception as e:
        print(f"筛选产品时发生错误: {e}")
        return False

def update_product_price(filename, product_id, new_price):
    """更新产品价格"""
    try:
        if not os.path.exists(filename):
            print(f"文件 {filename} 不存在！")
            return False
        
        # 读取所有数据
        rows = []
        headers = None
        updated = False
        
        with open(filename, "r", encoding="utf-8") as file:
            reader = csv.DictReader(file)
            headers = reader.fieldnames
            
            for row in reader:
                if row["产品ID"] == product_id:
                    old_price = row["价格"]
                    row["价格"] = new_price
                    print(f"已将产品 {product_id} 的价格从 ${old_price} 更新为 ${new_price}")
                    updated = True
                
                rows.append(row)
        
        if not updated:
            print(f"没有找到产品ID为 '{product_id}' 的产品！")
            return False
        
        # 写回更新后的数据
        with open(filename, "w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=headers)
            writer.writeheader()
            writer.writerows(rows)
        
        return True
    
    except Exception as e:
        print(f"更新产品价格时发生错误: {e}")
        return False
